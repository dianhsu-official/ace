#include <bits/stdc++.h>
using namespace std;

int n,x,sum,temp,ans;
int a[110];

int main()	{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n >> x;
	for(int i=1 ; i<n ; i++)	cin >> a[i];
	sort(a+1 , a+n);
	for(int i=2 ; i<n ; i++)	sum += a[i];
	if(sum < x)	{
		cout << -1 << endl;
		return 0;
	}
	temp = sum;
	for(int i=0 ; i<=100 ; i++)	{
		if(i >= a[1] && i <= a[n-1])	temp = sum + i - a[n-1] ;
		if(temp == x)	{
			cout << i << endl ;
			return 0;
		}
	}
	return 0;
}
